    ```html
      <vscode-option>Afghanistan</vscode-option>
      <vscode-option>Albania</vscode-option>
      ...
    </vscode-single-select>
    ```
  </Fragment>
  <Fragment slot="js">
    ```javascript
    const cb = document.querySelector("#combobox-example");
    const sl = document.querySelector("#search-method-selector");

    sl.addEventListener("change", () => {
      cb.filter = sl.options[sl.selectedIndex].value;
    });

    cb.filter = sl.options[sl.selectedIndex].value;
    ```
  </Fragment>
</Demo>

## Creatable

If `creatable` is enabled in combobox mode, the user can add new options to the component. If the
search field is not empty, an additional option will appear at the bottom of the list, allowing the
user to extend the existing options.

<Demo>
  <CreatableExample />
  <Fragment slot="html">
    ```html "combobox" "creatable"
    <vscode-single-select combobox creatable>
      <vscode-option>Afghanistan</vscode-option>
      <vscode-option>Albania</vscode-option>
      <vscode-option>Algeria</vscode-option>
      ...
    </vscode-single-select>
    ```
  </Fragment>
</Demo>
